330 void Canvas::OnLeftUp(wxMouseEvent& event) {
331 
332  // Holds which node and selection state occurred upon user's left button up
333  auto endSelection = Select(event.GetPosition());
334 
335  switch (m_selection.state) {
336 
337  // Finish move action

346  case Selection::State::NODE_OUTPUT:

350  m_incompleteEdge->ConnectDestination(m_nodes[endSelection]);

354  }
355  else
356  m_edges.erase(m_incompleteEdge->GetID());
357 
358  break;
359 
360  // Check that user selected an output to pair with the input and then connect
361  case Selection::State::NODE_INPUT:
362  if (endSelection.state == Selection::State::NODE_OUTPUT
363  && m_nodes[endSelection] != m_nodes[m_selection]) {
364 
365  m_incompleteEdge->ConnectSource(m_nodes[endSelection]);

441 // Capture mouse scrolling in order to zoom the camera
442 void Canvas::OnMouseWheel(wxMouseEvent& event) {
443 
444  // Convert mouse position from screen to world coordinates
445  wxPoint2DDouble mousePosition = event.GetPosition();
446  wxAffineMatrix2D inverse = GetCameraTransform();
447  inverse.Invert();
448  wxPoint2DDouble worldMousePosition = inverse.TransformPoint(mousePosition);
449 
450  // Determine the zoom scale factor
451  double scaleFactor = pow(2, 0.1 * event.GetWheelRotation() / event.GetWheelDelta());
452 
453  // Adjust the zoom and translation of the camera
454  m_cameraZoom.Scale(scaleFactor, scaleFactor);
455  m_zoomLevel = scaleFactor;
456  m_cameraPan.Translate((1 - scaleFactor) * (worldMousePosition.m_x), (1 - scaleFactor) * (worldMousePosition.m_y));
457 
458  Refresh();
459 }
