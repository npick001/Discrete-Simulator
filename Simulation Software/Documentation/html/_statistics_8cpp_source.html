228 double StatisticsObject::NewtonsMethod(double alpha, double initialGuess)
229 {
230  /*
231  Need to find the quantile of a chi-squared distribution given a given alpha level.
232  So, we are looking for a value x such that CDF(x) = alpha.
233  This can be framed as a root-finding problem: find the root of the function f(x) = CDF(x) - alpha.
234  */
235 
236  /*
237  The Newton-Raphson method is a root-finding algorithm that requires a function and its derivative.
238  In your case, you want to find the critical value x for which the CDF of the chi-square distribution equals 1 - alpha.
239  This means that the function you're looking for the root of is F(x) = CDF(x) - (1 - alpha).
240  */
241 
242  /*
243 
244  x_(n+1) = x_n - f(x_n) / f'(x_n)
245 
246  */
247  return 0;
248 }

250 double StatisticsObject::ChiSquareCriticalValue(double alpha, double dof) {
251 
252  // Use newton iteration to approximate the critical value
253 
254  double x = 2.0 * dof; // A reasonable initial guess
255  int max_iters = 250000;
256 
257  for (int i = 0; i < max_iters; i++) {
258  double cdf = ChiSquareCDF(dof, x, 1000);
259  double pdf_value = PDF_Chi_Square(x, dof);
260  double f = cdf - (1 - alpha);
261  double new_x = x - f / pdf_value;
262 
263  // If the change is small enough, we've converged
264  if (abs(new_x - x) < m_iterationTolerance) {
265  return new_x;
266  }
267 
268  x = new_x;
269  }
270 
271  // If we reach here, we did not converge in the given number of iterations
272  return x;
273 }

625 double StatisticsObject::ChiSquareCDF(double dof, double x_upper, int steps)
626 {
627  // Define step size
628  double stepSize = x_upper / steps;
629 
630  // Initialize area
631  double area = 0.0;
632 
633  // Apply Simpson's Rule
634  for (int i = 0; i <= steps; i++) {
635  double x = i * stepSize;
636  double multiplier = 1.0;
637 
638  // Check if it's the first point or the last point
639  if (i == 0 || i == steps)
640  multiplier = 1;
641  // Check if it's an odd or even step
642  else if (i % 2 == 1)
643  multiplier = 4;
644  else if (i % 2 == 0)
645  multiplier = 2;
646 
647  double y = PDF_Chi_Square(x, dof);
648 
649  area += multiplier * y;
650  }
651 
652  // Multiply the result by h/3
653  area *= (stepSize / 3.0);
654 
655  if (area > 1.0) {
656  std::cout << "\nERROR IN INTEGRATING CHI_SQUARE PDF. AREA > 1." << std::endl;
657  std::cout << "CALCULATED AREA: " << area << std::endl << std::endl;
658  return -1.0;
659  }
660 
661  return area;
662 }
