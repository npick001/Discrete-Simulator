167 // Returns the selection state of the component given where the user clicked
168 Selection GraphicalNode::Select(const wxAffineMatrix2D& camera, wxPoint2DDouble clickPosition) {
169 
170  // Transform click position from window coordinates to node's local coordinates
171  auto windowToLocal = camera;
172  windowToLocal.Concat(GetTransform());
173  windowToLocal.Invert();
174  clickPosition = windowToLocal.TransformPoint(clickPosition);
175 
176  // Return selection state according to what user clicked on
177  if (m_inputRect.Contains(clickPosition))
178  return { this, Selection::State::NODE_INPUT };
179  else if (m_outputRect.Contains(clickPosition))
180  return { this, Selection::State::NODE_OUTPUT };
181  else if (m_bodyShape.Contains(clickPosition))
182  return { this, Selection::State::NODE };
183  else
184  return { nullptr, Selection::State::NONE };
185 }
